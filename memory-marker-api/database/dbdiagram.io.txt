Table marker {
  id int [pk, increment] // auto-increment
  title varchar
  description varchar
  url varchar
  createdAt timestamp
  updatedAt timestamp
}

Table topic {
  id int [pk, increment]
  name varchar
  description varchar
  createdAt timestamp
  updatedAt timestamp
 }
 
 Table marker_topic {
   marker_id int [ref: > marker.id]
   topic_id int [ref: > topic.id]
   createdAt timestamp
 }
